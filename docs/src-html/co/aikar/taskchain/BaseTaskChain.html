159    /**
160     * Finished adding tasks, begins executing them with a done notifier and error handler
161     * @param done The Callback to handle when the chain has finished completion. Argument to consumer contains finish state
162     * @param errorHandler The Error handler to handle exceptions
163     */
164    public void execute(Consumer<Boolean> done, BiConsumer<Exception, TaskChainTasks.Task<?, ?>> errorHandler) {
165        if (errorHandler == null) {
166            errorHandler = factory.getDefaultErrorHandler();
167        }
168        this.doneCallback = done;
169        this.errorHandler = errorHandler;
170        execute0();
171    }

175    <A1, A2, A3> void handleAbortAction(TaskChainAbortAction<A1, A2, A3> action, A1 arg1, A2 arg2, A3 arg3) {
176        if (action != null) {
177            final TaskChain<?> prev = currentChain.get();
178            try {
179                currentChain.set(this);
180                action.onAbort(this, arg1, arg2, arg3);
181            } catch (Exception e) {
182                TaskChainUtil.logError("TaskChain Exception in Abort Action handler: " + action.getClass().getName());
183                TaskChainUtil.logError("Current Action Index was: " + currentActionIndex);
184                e.printStackTrace();
185            } finally {
186                currentChain.set(prev);
187            }
188        }
189        TaskChain.abort();
190    }
191
192    void execute0() {
193        synchronized (this) {
194            if (this.executed) {
195                throw new RuntimeException("Already executed");
196            }
197            this.executed = true;
198        }
199        async = !impl.isMainThread();
200        nextTask();
201    }
202
203    void done(boolean finished) {
204        this.done = true;
205        if (this.doneCallback != null) {
206            final TaskChain<?> prev = currentChain.get();
207            try {
208                currentChain.set(this);
209                this.doneCallback.accept(finished);
210            } catch (Exception e) {
211                this.handleError(e, null);
212            } finally {
213                currentChain.set(prev);
214            }
215        }
216    }

218    <R> TaskChain<R> add0(Boolean async, TaskChainTasks.Task task) {
219        //noinspection unchecked
220        return add0(new TaskHolder<>(this, async, task));
221    }

223    @SuppressWarnings({"rawtypes", "WeakerAccess"})
224    <R> TaskChain<R> add0(TaskHolder<R,?> task) {
225        synchronized (this) {
226            if (this.executed) {
227                throw new RuntimeException("TaskChain is executing");
228            }
229        }
230
231        this.chainQueue.add(task);
232        //noinspection unchecked
233        return (TaskChain<R>) this;
234    }
235
236    /**
237     * Fires off the next task, and switches between Async/Sync as necessary.
238     */
239    private void nextTask() {
240        synchronized (this) {
241            this.currentHolder = this.chainQueue.poll();
242            if (this.currentHolder == null) {
243                this.done = true; // to ensure its done while synchronized
244            }
245        }
246
247        if (this.currentHolder == null) {
248            this.previous = null;
249            // All Done!
250            this.done(true);
251            return;
252        }
253
254        Boolean isNextAsync = this.currentHolder.async;
255        if (isNextAsync == null || factory.shutdown) {
256            this.currentHolder.run();
257        } else if (isNextAsync) {
258            if (this.async) {
259                this.currentHolder.run();
260            } else {
261                impl.postAsync(() -> {
262                    this.async = true;
263                    this.currentHolder.run();
264                });
265            }
266        } else {
267            if (this.async) {
268                impl.postToMain(() -> {
269                    this.async = false;
270                    this.currentHolder.run();
271                });
272            } else {
273                this.currentHolder.run();
274            }
275        }
276    }
277
278    private void handleError(Throwable throwable, TaskChainTasks.Task<?, ?> task) {
279        Exception e = throwable instanceof Exception ? (Exception) throwable : new Exception(throwable);
280        if (errorHandler != null) {
281            final TaskChain<?> prev = currentChain.get();
282            try {
283                currentChain.set(this);
284                errorHandler.accept(e, task);
285            } catch (Exception e2) {
286                TaskChainUtil.logError("TaskChain Exception in the error handler!" + e2.getMessage());
287                TaskChainUtil.logError("Current Action Index was: " + currentActionIndex);
288                e.printStackTrace();
289            } finally {
290                currentChain.set(prev);
291            }
292        } else {
293            TaskChainUtil.logError("TaskChain Exception on " + (task != null ? task.getClass().getName() : "Done Hander") + ": " + e.getMessage());
294            TaskChainUtil.logError("Current Action Index was: " + currentActionIndex);
295            e.printStackTrace();
296        }
297    }
298
299    private void abortChain() {
300        this.previous = null;
301        this.chainQueue.clear();
302        this.done(false);
303    }

305    <R> CompletableFuture<List<R>> getFuture(List<CompletableFuture<R>> futures) {
306        CompletableFuture<List<R>> onDone = new CompletableFuture<>();
307        CompletableFuture<?>[] futureArray = new CompletableFuture<?>[futures.size()];
308        CompletableFuture.allOf((CompletableFuture<?>[]) futures.toArray(futureArray)).whenComplete((aVoid, throwable) -> {
309            if (throwable != null) {
310                onDone.completeExceptionally(throwable);
311            } else {
312                boolean[] error = {false};
313                final List<R> results = futures.stream().map(f -> {
314                    try {
315                        return f.join();
316                    } catch (Exception e) {
317                        error[0] = true;
318                        BaseTaskChain.this.handleError(e, BaseTaskChain.this.currentHolder.task);
319                        return null;
320                    }
321                }).collect(Collectors.toList());
322                if (error[0]) {
323                    onDone.completeExceptionally(new Exception("Future Dependant had an exception"));
324                } else {
325                    onDone.complete(results);
326                }
327            }
328        });
329        return onDone;
330    }

335    /**
336     * Provides foundation of a task with what the previous task type should return
337     * to pass to this and what this task will return.
338     * @param <R> Return Type
339     * @param <A> Argument Type Expected
340     */
341    @SuppressWarnings("AccessingNonPublicFieldOfAnotherObject")
342    private static class TaskHolder<R, A> {
343        private final BaseTaskChain<A> chain;
344        private final TaskChainTasks.Task<R, A> task;
345        final Boolean async;
346
347        private boolean executed = false;
348        private boolean aborted = false;
349        private final int actionIndex;
350
351        TaskHolder(BaseTaskChain<A> chain, Boolean async, TaskChainTasks.Task<R, A> task) {
352            this.actionIndex = chain.actionIndex++;
353            this.task = task;
354            this.chain = chain;
355            this.async = async;
356        }
357
358        /**
359         * Called internally by Task Chain to facilitate executing the task and then the next task.
360         */
361        private void run() {
362            final Object arg = this.chain.previous;
363            this.chain.previous = null;
364            this.chain.currentActionIndex = this.actionIndex;
365            final TaskChain<?> prevChain = currentChain.get();
366            try {
367                currentChain.set(this.chain);
368                if (this.task instanceof TaskChainTasks.FutureTask) {
369                    //noinspection unchecked
370                    final CompletableFuture<R> future = ((TaskChainTasks.FutureTask<R, A>) this.task).runFuture((A) arg);
371                    if (future == null) {
372                        throw new NullPointerException("Must return a Future");
373                    }
374                    future.whenComplete((r, throwable) -> {
375                        if (throwable != null) {
376                            this.chain.handleError(throwable, this.task);
377                            this.abort();
378                        } else {
379                            this.next(r);
380                        }
381                    });
382                } else if (this.task instanceof TaskChainTasks.AsyncExecutingTask) {
383                    //noinspection unchecked
384                    ((TaskChainTasks.AsyncExecutingTask<R, A>) this.task).runAsync((A) arg, this::next);
385                } else {
386                    //noinspection unchecked
387                    next(this.task.run((A) arg));
388                }
389            } catch (Throwable e) {
390                //noinspection ConstantConditions
391                if (e instanceof AbortChainException) {
392                    this.abort();
393                    return;
394                }
395                this.chain.handleError(e, this.task);
396                this.abort();
397            } finally {
398                if (prevChain != null) {
399                    currentChain.set(prevChain);
400                } else {
401                    currentChain.remove();
402                }
403            }
404        }
405
406        /**
407         * Abort the chain, and clear tasks for GC.
408         */
409        private synchronized void abort() {
410            this.aborted = true;
411            this.chain.abortChain();
412        }
413
414        /**
415         * Accepts result of previous task and executes the next
416         */
417        private void next(Object resp) {
418            synchronized (this) {
419                if (this.aborted) {
420                    this.chain.done(false);
421                    return;
422                }
423                if (this.executed) {
424                    this.chain.done(false);
425                    throw new RuntimeException("This task has already been executed.");
426                }
427                this.executed = true;
428            }
429
430            this.chain.async = !this.chain.impl.isMainThread(); // We don't know where the task called this from.
431            this.chain.previous = resp;
432            this.chain.nextTask();
433        }
434    }
